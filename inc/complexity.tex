\subsection{計算量}\frame{  \frametitle{計算量}  \begin{block}{ランダウの記号}    \[f(x) \in O(g(x)) \Leftrightarrow \lim_{x \rightarrow \infty}{\frac{f(x)}{g(x)} < \infty} \]    \[f(x) \in \Omega(g(x)) \Leftrightarrow \lim_{x \rightarrow \infty}{\frac{f(x)}{g(x)} > 0} \]    \[f(x) \in \Theta(g(x)) \Leftrightarrow 0 < \lim_{x \rightarrow \infty}{\frac{f(x)}{g(x)} < \infty} \]    \[f(x) \in o(g(x)) \Leftrightarrow \lim_{x \rightarrow \infty}{\frac{f(x)}{g(x)} = 0} \]  \end{block}}\frame{  \center{    \includegraphics<1>[width=8cm]{image/complexity01.pdf}    \includegraphics<2>[width=8cm]{image/complexity02.pdf}  }}\frame{  \frametitle{計算量}  \begin{alertblock}{ランダウの記号}    \center{      $f(x) \in O(g(x))$ を $f(x) = O(g(x))$ と書く事がある！！\\    }  \end{alertblock}  \begin{itemize}  \item いろいろと便利  \item しかしこれはどうなんだ  \end{itemize}}
